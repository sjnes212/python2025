x=-5
y=2
print(x//y)
>> -3
# *** -5/2 =-2.5  ***
# *** floor of a negative number you get one smaller number with no dots  ***
# *** -3 -2 -1 0 1 2 3  ***

x=5.0
y=2
print(x//y)
>> 2.0
# *** 5/2 =2.5  ***
# *** floor of a positive number you get smaller number with float  ***
# *** -3 -2 -1 0 1 2 3  ***

x=2
y=-2
print(x**y)
>> 0.25
# *** 2^-2 =1/4 =0.25  ***
# *** is used for power operator  ***
